# useEasings

Easings as functions

## Usage

```ts
import { useEasings } from '@maas/vue-composables'

const {
  linear,
  easeInQuad,
  easeOutQuad,
  easeInOutQuad,
  easeInCubic,
  easeOutCubic,
  easeInOutCubic,
  easeInQuart,
  easeOutQuart,
  easeInOutQuart,
  easeInQuint,
  easeOutQuint,
  easeInOutQuint,
} = useEasings()

easeOutQuad(300)
```
